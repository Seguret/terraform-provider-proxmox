# proxmox_virtual_environment_metrics_server (Resource)

Manages a Proxmox VE external metrics server (Graphite or InfluxDB).

## Example Usage

### Graphite Server

```terraform
resource "proxmox_virtual_environment_metrics_server" "graphite" {
  name    = "graphite-server"
  type    = "graphite"
  server  = "graphite.example.com"
  port    = 2003
  enabled = true
}
```

### InfluxDB Server (HTTP)

```terraform
resource "proxmox_virtual_environment_metrics_server" "influxdb" {
  name             = "influxdb-http"
  type             = "influxdb"
  server           = "influxdb.example.com"
  port             = 8086
  influxdb_proto   = "http"
  bucket           = "proxmox"
  enabled          = true
}
```

### InfluxDB Server (HTTPS)

```terraform
resource "proxmox_virtual_environment_metrics_server" "influxdb_secure" {
  name             = "influxdb-https"
  type             = "influxdb"
  server           = "influxdb.example.com"
  port             = 8086
  influxdb_proto   = "https"
  bucket           = "proxmox-metrics"
  max_body_size    = 25000000
  enabled          = true
}
```

### InfluxDB UDP Server

```terraform
resource "proxmox_virtual_environment_metrics_server" "influxdb_udp" {
  name           = "influxdb-udp"
  type           = "influxdb"
  server         = "influxdb.example.com"
  port           = 8089
  influxdb_proto = "udp"
  bucket         = "proxmox"
  mtu            = 1500
  enabled        = true

```
