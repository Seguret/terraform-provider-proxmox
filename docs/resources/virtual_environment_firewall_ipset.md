# proxmox_virtual_environment_firewall_ipset (Resource)

Manages a Proxmox VE firewall IP set.

## Example Usage

### Cluster IP Set

```terraform
resource "proxmox_virtual_environment_firewall_ipset" "trusted_ips" {
  scope   = "cluster"
  name    = "trusted-networks"
  comment = "Trusted networks and IPs"
  
  cidrs = [
    {
      cidr    = "203.0.113.0/24"
      comment = "Office network"
    },
    {
      cidr    = "192.168.1.0/24"
      comment = "VPN subnet"
    },
    {
      cidr    = "10.0.0.50/32"
      comment = "Management server"
    }
  ]
}
```

### Node-Level IP Set

```terraform
resource "proxmox_virtual_environment_firewall_ipset" "node_whitelist" {
  scope   = "node/pve"
  name    = "node-access"
  comment = "Allowed IPs for node access"
  
  cidrs = [
    {
      cidr    = "203.0.113.10/32"
      comment = "Monitoring server"
    },
    {
      cidr    = "203.0.113.20/32"
      comment = "Backup server"
    }
  ]
}
```

### VM-Specific IP Set

```terraform
resource "proxmox_virtual_environment_firewall_ipset" "app_blacklist" {
  scope   = "vm/pve/100"
  name    = "blocked-ips"
  comment = "Blocked IPs for this VM"
  
  cidrs = [
    {
      cidr    = "192.0.2.0/24"
      comment = "Known malicious range"
    },
    {
      cidr    = "198.51.100.0/24"
      comment = "Threat intel range"
    }
  ]
}
```

### Dynamic IP Set

```terraform
resource "proxmox_virtual_environment_firewall_ipset" "datacenter_nets" {
  scope   = "cluster"
  name    = "dc-subnets"
  comment = "Datacenter subnets"
  
  cidrs = [
    { cidr = "10.0.0.0/24", comment = "Production" },
    { cidr = "10.0.1.0/24", comment = "Staging" },
    { cidr = "10.0.2.0/24", comment = "Development" },
    { cidr = "10.0.3.0/24", comment = "Management" },
  ]
}
```
